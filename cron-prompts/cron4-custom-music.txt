async function uploadCustomMusic(scrapbookId: string, file: File): Promise<string> {
    const supabase = createClient();
    const ext = file.name.split('.').pop() || 'mp3';
    const fileName = `music/${scrapbookId}-${Date.now()}.${ext}`;
    const arrayBuffer = await file.arrayBuffer();
    const fileData = new Uint8Array(arrayBuffer);

    const { error } = await supabase.storage
        .from('photos')
        .upload(fileName, fileData, {
            contentType: file.type || 'audio/mpeg',
            cacheControl: '3600',
        });

    if (error) throw error;

    const { data: { publicUrl } } = supabase.storage.from('photos').getPublicUrl(fileName);
    return publicUrl;
}

async function uploadCustomMusic(file: File): Promise<string> {
    const supabase = createClient();
    const ext = file.name.split('.').pop()?.toLowerCase() || 'mp3';
    const fileName = `music/${Date.now()}-${Math.random().toString(36).slice(2, 8)}.${ext}`;
    const arrayBuffer = await file.arrayBuffer();
    const fileData = new Uint8Array(arrayBuffer);

    const { error } = await supabase.storage
        .from('photos')
        .upload(fileName, fileData, {
            contentType: file.type || 'audio/mpeg',
            cacheControl: '3600',
        });

    if (error) throw error;

    const { data: { publicUrl } } = supabase.storage.from('photos').getPublicUrl(fileName);
    return publicUrl;
}
